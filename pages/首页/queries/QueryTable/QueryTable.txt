SELECT IDay, GDay, COALESCE(ICount, 0) AS ICount, COALESCE(GCount, 0) AS GCount
FROM
(
  SELECT DATE_FORMAT(PHY_EXAM_DATE, '%Y%m%d') AS IDay, COALESCE(COUNT(*), 0) AS ICount
  FROM PERSON_REG_INFO
  WHERE DATE_FORMAT(PHY_EXAM_DATE, '%Y') >= '2023'
  AND REG_TYPE = 1
  GROUP BY DATE_FORMAT(PHY_EXAM_DATE, '%Y%m%d')
  ORDER BY DATE_FORMAT(PHY_EXAM_DATE, '%Y%m%d')
) AS I
LEFT JOIN
(
  SELECT DATE_FORMAT(PHY_EXAM_DATE, '%Y%m%d') AS GDay, COALESCE(COUNT(*), 0) AS GCount
  FROM PERSON_REG_INFO
  WHERE DATE_FORMAT(PHY_EXAM_DATE, '%Y') >= '2023'
  AND REG_TYPE = 2
  GROUP BY DATE_FORMAT(PHY_EXAM_DATE, '%Y%m%d')
  ORDER BY DATE_FORMAT(PHY_EXAM_DATE, '%Y%m%d')
) AS G
ON IDay = GDay
ORDER BY IDay;
