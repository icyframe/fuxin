SELECT
  t2.PHY_EXAM_COMBINE_NAME AS ZHMC,
  t2.ID AS zhxmId,
  t1.ACTUAL_PRICE,
  t1.DISCOUNT,
  t1.STANDARD_PRICE,
  t1.PAY_STATUS,
  t1.PAY_MODE,
  t1.ADD_ITEM_FLAG AS ADD_FLAG,
  t1.CHECK_DOCTOR,
  t1.CHECK_STATUS,
  t3.PHY_EXAM_PACKAGE_NAME AS TCMC,
  t4.DEPT_NAME,
  t4.ID AS DEPARTMENT_ID,
  t4.DEPT_ADDRESS,
  t4.REMARK AS DEPT_REMARK,
  t2.PHY_EXAM_TYPE
FROM
  REG_ITEM_INFO t1
LEFT JOIN
  DICT_PHY_EXAM_COMBINE t2 ON t1.ITEM_ID = t2.ID
LEFT JOIN
  DICT_PHY_EXAM_PACKAGE t3 ON t1.PKG_ID = t3.ID
LEFT JOIN
  DICT_PHY_EXAM_DEPT t4 ON t4.ID = t2.DEPT_ADDRESS
WHERE
  t1.REG_ID = CAST({{ appsmith.URL.queryParams.PE_ID }} AS CHAR)
  AND t1.DELETE_FLAG = 0
ORDER BY t1.PKG_ID ASC;
