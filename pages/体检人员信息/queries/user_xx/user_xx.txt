SELECT *
FROM (
  SELECT
    t1.ID,
    t1.FILE_NO,
    t1.NAME,
    t1.PHY_EXAM_NO,
    t1.REG_SERIAL_NO,
    t1.ID_CARD_TYPE,
    t1.ID_CARD_NO,
    t1.RECOMMEND_DEPT,
    t1.RECOMMENDER,
    DATE_FORMAT(t1.PHY_EXAM_DATE, '%Y-%m-%d %H:%i:%s') AS PE_DATE,
    DATE_FORMAT(t1.BIRTHDAY, '%Y-%m-%d %H:%i:%s') AS BIRTHDAY,
    t1.AGE,
    t1.GENDER,
    t1.MARRIAGE,
    t4.PHY_BUSINESS_TYPE_NAME AS ywlx,
    t1.PHONE_NO,
    t3.PHY_EXAM_TYPE_NAME AS tjlx,
    t5.PERSONNEL_TYPE_NAME AS rylx,
    t1.ADDRESS,
    t1.EMAIL,
    t1.NATION,
    ct.CARD_TYPE_NAME,
    dk.DEPT_NAME AS KSMC,
    t1.REG_TIME,
    t1.BIZ_TYPE_ID,
    t1.PHY_EXAM_TYPE_ID,
    t1.PERSON_TYPE_ID
  FROM
    PERSON_REG_INFO t1
  LEFT JOIN
    TEAM_GROUP_INFO t2 ON t1.GROUP_ID = t2.ID
  LEFT JOIN
    PERSON_CARD_TYPE ct ON t1.ID_CARD_TYPE = ct.ID
  LEFT JOIN
    DICT_PHY_EXAM_TYPE t3 ON t1.PHY_EXAM_TYPE_ID = t3.ID
  LEFT JOIN
    DICT_PHY_BUSINESS_TYPE t4 ON t1.BIZ_TYPE_ID = t4.ID
  LEFT JOIN
    DICT_USER_TYPE t5 ON t1.PERSON_TYPE_ID = t5.ID
  LEFT JOIN
    DICT_PHY_EXAM_PACKAGE t6 ON t1.PKG_ID = t6.ID
  LEFT JOIN
    DICT_PHY_EXAM_DEPT dk ON t1.RECOMMEND_DEPT = dk.ID
  ORDER BY
    t1.PHY_EXAM_NO DESC
) AS subquery
WHERE
  ID = CAST({{ appsmith.URL.queryParams.PE_ID }} AS CHAR);
